-- restxlib by @твой_ник (простая UI-lib в стиле чит-клиента)
local RestXLib = {}
RestXLib.__index = RestXLib

local TweenService = game:GetService("TweenService")
local UserInputService = game:GetService("UserInputService")

local function createInstance(class, props)
    local inst = Instance.new(class)
    for prop, value in pairs(props or {}) do
        inst[prop] = value
    end
    return inst
end

-- Основная функция создания окна
function RestXLib:CreateWindow(title: string, accentColor: Color3?)
    local self = setmetatable({}, RestXLib)
    
    accentColor = accentColor or Color3.fromRGB(170, 85, 255) -- фиолетовый по умолчанию
    
    -- Главный ScreenGui
    local ScreenGui = createInstance("ScreenGui", {
        Name = "RestXLib",
        Parent = game.Players.LocalPlayer:WaitForChild("PlayerGui"),
        ResetOnSpawn = false,
        ZIndexBehavior = Enum.ZIndexBehavior.Sibling
    })
    
    -- Главное окно
    local MainFrame = createInstance("Frame", {
        Name = "Main",
        Parent = ScreenGui,
        Size = UDim2.new(0, 520, 0, 380),
        Position = UDim2.new(0.5, -260, 0.5, -190),
        BackgroundColor3 = Color3.fromRGB(20, 20, 25),
        BorderSizePixel = 0
    })
    
    local UICorner = createInstance("UICorner", {Parent = MainFrame, CornerRadius = UDim.new(0, 8)})
    local UIStroke = createInstance("UIStroke", {Parent = MainFrame, Color = Color3.fromRGB(40, 40, 50), Thickness = 1})
    
    -- Заголовок (draggable)
    local TitleBar = createInstance("Frame", {
        Name = "TitleBar",
        Parent = MainFrame,
        Size = UDim2.new(1, 0, 0, 40),
        BackgroundColor3 = Color3.fromRGB(15, 15, 20),
        BorderSizePixel = 0
    })
    createInstance("UICorner", {Parent = TitleBar, CornerRadius = UDim.new(0, 8)})
    
    local TitleLabel = createInstance("TextLabel", {
        Name = "Title",
        Parent = TitleBar,
        Size = UDim2.new(1, -100, 1, 0),
        BackgroundTransparency = 1,
        Text = "  " .. title,
        TextColor3 = Color3.fromRGB(255, 255, 255),
        TextXAlignment = Enum.TextXAlignment.Left,
        Font = Enum.Font.GothamBold,
        TextSize = 16
    })
    
    -- Кнопка закрытия
    local CloseButton = createInstance("TextButton", {
        Name = "Close",
        Parent = TitleBar,
        Size = UDim2.new(0, 30, 0, 30),
        Position = UDim2.new(1, -35, 0, 5),
        BackgroundTransparency = 1,
        Text = "✕",
        TextColor3 = Color3.fromRGB(255, 80, 80),
        TextSize = 18,
        Font = Enum.Font.GothamBold
    })
    
    -- Draggable
    local dragging = false
    local dragInput, dragStart, startPos
    
    TitleBar.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            dragging = true
            dragStart = input.Position
            startPos = MainFrame.Position
        end
    end)
    
    UserInputService.InputChanged:Connect(function(input)
        if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
            local delta = input.Position - dragStart
            MainFrame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
        end
    end)
    
    UserInputService.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            dragging = false
        end
    end)
    
    CloseButton.MouseButton1Click:Connect(function()
        ScreenGui:Destroy()
    end)
    
    self.MainFrame = MainFrame
    self.AccentColor = accentColor
    self.Tabs = {}
    
    -- Контейнер для табов и контента
    self.TabContainer = createInstance("Frame", {
        Name = "TabContainer",
        Parent = MainFrame,
        Size = UDim2.new(1, 0, 1, -40),
        Position = UDim2.new(0, 0, 0, 40),
        BackgroundTransparency = 1
    })
    
    return self
end

-- Добавление таба
function RestXLib:AddTab(tabName: string)
    -- Здесь будет горизонтальная панель табов + Frame для контента
    -- Для простоты сначала сделаем один видимый контейнер, потом добавим переключение
    
    local TabButton = createInstance("TextButton", {
        Parent = self.TabContainer, -- позже сделаем отдельный TabBar
        Size = UDim2.new(0, 100, 0, 30),
        BackgroundColor3 = Color3.fromRGB(30, 30, 35),
        Text = tabName,
        TextColor3 = Color3.fromRGB(220, 220, 220),
        Font = Enum.Font.Gotham,
        TextSize = 14
    })
    createInstance("UICorner", {Parent = TabButton})
    
    local ContentFrame = createInstance("ScrollingFrame", {
        Name = tabName .. "Content",
        Parent = self.TabContainer,
        Size = UDim2.new(1, 0, 1, 0),
        BackgroundTransparency = 1,
        ScrollBarThickness = 4,
        BorderSizePixel = 0
    })
    
    local UIListLayout = createInstance("UIListLayout", {
        Parent = ContentFrame,
        SortOrder = Enum.SortOrder.LayoutOrder,
        Padding = UDim.new(0, 8)
    })
    
    ContentFrame.CanvasSize = UDim2.new(0, 0, 0, 0)
    UIListLayout:GetPropertyChangedSignal("AbsoluteContentSize"):Connect(function()
        ContentFrame.CanvasSize = UDim2.new(0, 0, 0, UIListLayout.AbsoluteContentSize.Y + 20)
    end)
    
    table.insert(self.Tabs, {Button = TabButton, Content = ContentFrame})
    
    -- Простое переключение (пока только последний таб видим)
    for _, tab in ipairs(self.Tabs) do
        tab.Content.Visible = false
    end
    ContentFrame.Visible = true
    
    return {
        AddGroupbox = function(groupName)
            local Group = createInstance("Frame", {
                Parent = ContentFrame,
                Size = UDim2.new(1, -20, 0, 40), -- авто-размер потом
                BackgroundColor3 = Color3.fromRGB(25, 25, 30),
                BorderSizePixel = 0
            })
            createInstance("UICorner", {Parent = Group, CornerRadius = UDim.new(0, 6)})
            
            local GroupTitle = createInstance("TextLabel", {
                Parent = Group,
                Size = UDim2.new(1, 0, 0, 30),
                BackgroundTransparency = 1,
                Text = "  " .. groupName,
                TextColor3 = self.AccentColor,
                TextXAlignment = Enum.TextXAlignment.Left,
                Font = Enum.Font.GothamBold,
                TextSize = 15
            })
            
            local GroupContent = createInstance("Frame", {
                Parent = Group,
                Size = UDim2.new(1, -10, 1, -35),
                Position = UDim2.new(0, 5, 0, 35),
                BackgroundTransparency = 1
            })
            
            local List = createInstance("UIListLayout", {Parent = GroupContent, Padding = UDim.new(0, 6)})
            
            return {
                AddToggle = function(name, default, callback)
                    -- Реализация Toggle ниже
                    local ToggleFrame = createInstance("Frame", {Parent = GroupContent, Size = UDim2.new(1, 0, 0, 30), BackgroundTransparency = 1})
                    
                    local ToggleLabel = createInstance("TextLabel", {
                        Parent = ToggleFrame,
                        Size = UDim2.new(0.7, 0, 1, 0),
                        BackgroundTransparency = 1,
                        Text = name,
                        TextColor3 = Color3.fromRGB(255,255,255),
                        TextXAlignment = Enum.TextXAlignment.Left
                    })
                    
                    local ToggleButton = createInstance("TextButton", {
                        Parent = ToggleFrame,
                        Size = UDim2.new(0, 40, 0, 20),
                        Position = UDim2.new(1, -50, 0.5, -10),
                        BackgroundColor3 = default and self.AccentColor or Color3.fromRGB(50,50,55),
                        Text = "",
                        AutoButtonColor = false
                    })
                    createInstance("UICorner", {Parent = ToggleButton, CornerRadius = UDim.new(1,0)})
                    
                    local enabled = default
                    
                    ToggleButton.MouseButton1Click:Connect(function()
                        enabled = not enabled
                        TweenService:Create(ToggleButton, TweenInfo.new(0.2), {BackgroundColor3 = enabled and self.AccentColor or Color3.fromRGB(50,50,55)}):Play()
                        if callback then callback(enabled) end
                    end)
                    
                    return {Set = function(val) enabled = val; /* обновить цвет */ end}
                end,
                
                AddButton = function(name, callback)
                    local Btn = createInstance("TextButton", {
                        Parent = GroupContent,
                        Size = UDim2.new(1, 0, 0, 35),
                        BackgroundColor3 = Color3.fromRGB(40,40,45),
                        Text = name,
                        TextColor3 = Color3.fromRGB(255,255,255),
                        Font = Enum.Font.Gotham
                    })
                    createInstance("UICorner", {Parent = Btn})
                    
                    Btn.MouseButton1Click:Connect(callback or function() end)
                end,
                
                -- Добавь сюда Slider, Label и т.д.
            }
        end
    }
end

-- Пример использования в конце файла для теста
return RestXLib
