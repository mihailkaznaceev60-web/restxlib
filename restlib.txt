-- Настройки ScrollingFrame
MainFrame.ScrollBarThickness = 8

-- //Close Frame

CloseFrame.Name = "CloseFrame"
CloseFrame.Parent = MainScreen

-- //Close Btn

-- // MAIN UI MECHANICS
local function scanBP()
    local backpack = plr:WaitForChild("Backpack"):GetChildren()
    local buttonsPerRow = 3
    local buttonWidth = 0.31
    local buttonHeight = 0.08
    local startX = 0.01
    local startY = 0.01
    local spacingX = 0.01
    local spacingY = 0.02
    
    local index = 0
    for _, child in ipairs(MainFrame:GetChildren()) do
        if child:IsA("TextButton") then
            child:Destroy()
        end
    end
    

    for _, item in ipairs(backpack) do
        if not item:FindFirstChild("TopPart") then
            index = index + 1
            
            local row = math.floor((index - 1) / buttonsPerRow)
            local col = (index - 1) % buttonsPerRow
            
            local posX = startX + col * (buttonWidth + spacingX)
            local posY = startY + row * (buttonHeight + spacingY)
            
            local itembtn = Instance.new("TextButton")
            itembtn.Parent = MainFrame
            itembtn.Text = item.Name
            itembtn.Size = UDim2.new(buttonWidth, 0, buttonHeight, 0)
            itembtn.Position = UDim2.new(posX, 0, posY, 0)
            itembtn.BackgroundColor3 = Color3.fromRGB(0, 255, 0)
            itembtn.TextColor3 = Color3.fromRGB(0, 0, 0)
            itembtn.TextScaled = true
            
            -- // CONNECTS
            itembtn.MouseButton1Click:Connect(function()
                weightstats.Text = "Weight:" .. item:GetAttribute("XP")
                goldstats.Text = "Gold: " .. tostring(item:GetAttribute("Golden"))
            end)
            itembtn.MouseButton2Click:Connect(function()
                if item:GetAttribute("Locked") == nil or item:GetAttribute("Locked") == false then
                    itembtn.BackgroundColor3 = Color3.new(255, 0, 0)
                    item:SetAttribute("Locked", true)
                else
                    itembtn.BackgroundColor3 = Color3.new(0, 255, 0)
                    item:SetAttribute("Locked", false)
                end
            end)
            
            -- // UIBTNSCORNER
            local itemuicorner = Instance.new("UICorner")
            itemuicorner.Parent = itembtn
            itemuicorner.CornerRadius = UDim.new(0, 8)
        end
    end
end
scanBP()
plr.Backpack.ChildAdded:Connect(scanBP)
plr.Backpack.ChildRemoved:Connect(scanBP)
